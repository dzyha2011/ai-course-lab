🔍 卷积神经网络基本原理

卷积神经网络（CNN）是一种特殊设计的深度神经网络，主要用于处理网格结构数据（如图像）。CNN通过三大核心特性实现高效的特征提取：

🎯 三大核心特性

🔍 局部感受野

模拟生物视觉系统，每个神经元仅与输入数据的局部区域连接，有效提取局部特征。

🔗 权值共享

同一卷积核在整个输入图像上重复使用，大幅减少参数数量，提高训练效率。

📉 空间降采样

通过池化操作减少特征图尺寸，降低计算复杂度并增强平移不变性。

🧮 核心数学公式

卷积操作： Y = X * W + b
其中 X 为输入特征图，W 为卷积核权重，b 为偏置，* 表示互相关运算
ReLU激活函数： ReLU(x) = max(0, x)
解决梯度消失问题，计算简单，收敛速度快
最大池化： Yi,j = maxu,v Xi×s+u, j×s+v
其中 s 为步长，保留区域内的最强响应

🏗️ CNN架构组件

组件 功能 关键参数 作用
卷积层 特征提取 卷积核数量、大小、步长、填充 提取局部特征，保持空间结构
激活层 非线性变换 激活函数类型（ReLU、Sigmoid等） 增强网络表达能力
池化层 降维采样 池化核大小、步长、类型 减少计算量，增强不变性
全连接层 分类决策 神经元数量、Dropout率 将特征映射到类别空间

def prepare_mnist_data():
    """准备MNIST数据集"""
    print("正在加载MNIST数据集...")
    mnist = fetch_openml('mnist_784', version=1, as_frame=False)
    X, y = mnist.data, mnist.target.astype(np.int32)
    
    # 数据预处理
    X = X.reshape(-1, 1, 28, 28) / 255.0  # 归一化并添加通道维度
    y = torch.tensor(y, dtype=torch.long)
    
    # 划分数据集
    X_train, X_test, y_train, y_test = train_test_split(
        X, y, test_size=10000, random_state=42, stratify=y
    )
    
    # 进一步划分训练集和验证集
    X_train, X_val, y_train, y_val = train_test_split(
        X_train, y_train, test_size=0.2, random_state=42, stratify=y_train
    )
    
    return X_train, X_val, X_test, y_train, y_val, y_test

class SimpleCNN(nn.Module):
    """简单的CNN模型"""
    def __init__(self):
        super(SimpleCNN, self).__init__()
        # 卷积层
        self.conv1 = nn.Conv2d(1, 32, kernel_size=3, padding=1)
        self.pool = nn.MaxPool2d(2, 2)
        self.conv2 = nn.Conv2d(32, 64, kernel_size=3, padding=1)
        
        # 全连接层
        self.fc1 = nn.Linear(64 * 7 * 7, 128)
        self.dropout = nn.Dropout(0.5)
        self.fc2 = nn.Linear(128, 10)
    
    def forward(self, x):
        # 第一层卷积 + 池化
        x = self.pool(F.relu(self.conv1(x)))  # (32, 14, 14)
        # 第二层卷积 + 池化
        x = self.pool(F.relu(self.conv2(x)))  # (64, 7, 7)
        # 展平
        x = x.view(-1, 64 * 7 * 7)
        # 全连接层
        x = F.relu(self.fc1(x))
        x = self.dropout(x)
        x = self.fc2(x)
        return x

🎯 训练函数

def train_model(model, train_loader, val_loader, num_epochs=5):
    """训练模型"""
    criterion = nn.CrossEntropyLoss()
    optimizer = torch.optim.Adam(model.parameters(), lr=0.001)
    scheduler = torch.optim.lr_scheduler.StepLR(optimizer, step_size=3, gamma=0.1)
    
    history = {'train_loss': [], 'train_acc': [], 'val_loss': [], 'val_acc': []}
    
    for epoch in range(num_epochs):
        # 训练阶段
        model.train()
        train_running_loss = 0.0
        train_correct = 0
        train_total = 0
        
        train_bar = tqdm(train_loader, desc=f'Epoch {epoch+1}/{num_epochs}')
        for inputs, labels in train_bar:
            inputs, labels = inputs.to(device), labels.to(device)
            
            # 前向传播
            optimizer.zero_grad()
            outputs = model(inputs)
            loss = criterion(outputs, labels)
            
            # 反向传播
            loss.backward()
            optimizer.step()
            
            # 统计
            train_running_loss += loss.item() * inputs.size(0)
            _, predicted = torch.max(outputs.data, 1)
            train_total += labels.size(0)
            train_correct += (predicted == labels).sum().item()
            
            # 更新进度条
            train_bar.set_postfix({
                'Loss': f'{loss.item():.4f}',
                'Acc': f'{100.*train_correct/train_total:.2f}%'
            })
        
        # 计算训练指标
        train_loss = train_running_loss / len(train_loader.dataset)
        train_acc = train_correct / train_total
        
        # 验证阶段
        model.eval()
        val_running_loss = 0.0
        val_correct = 0
        val_total = 0
        
        with torch.no_grad():
            for inputs, labels in val_loader:
                inputs, labels = inputs.to(device), labels.to(device)
                outputs = model(inputs)
                loss = criterion(outputs, labels)
                
                val_running_loss += loss.item() * inputs.size(0)
                _, predicted = torch.max(outputs.data, 1)
                val_total += labels.size(0)
                val_correct += (predicted == labels).sum().item()
        
        val_loss = val_running_loss / len(val_loader.dataset)
        val_acc = val_correct / val_total
        
        # 记录历史
        history['train_loss'].append(train_loss)
        history['train_acc'].append(train_acc)
        history['val_loss'].append(val_loss)
        history['val_acc'].append(val_acc)
        
        # 更新学习率
        scheduler.step()
        
        print(f'Epoch {epoch+1} - Train Loss: {train_loss:.4f}, Train Acc: {train_acc:.4f}, '
              f'Val Loss: {val_loss:.4f}, Val Acc: {val_acc:.4f}')
    
    return history

import torch
import torch.nn as nn
from torchvision import models

class CatDogResNet(nn.Module):
    def __init__(self, num_classes=2):
        super(CatDogResNet, self).__init__()
        # 加载预训练的ResNet50
        self.backbone = models.resnet50(pretrained=True)
        
        # 冻结前面的层
        for param in self.backbone.parameters():
            param.requires_grad = False
            
        # 替换分类头
        num_features = self.backbone.fc.in_features
        self.backbone.fc = nn.Sequential(
            nn.Dropout(0.5),
            nn.Linear(num_features, 512),
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(512, num_classes)
        )
        
    def forward(self, x):
        return self.backbone(x)

from torch.cuda.amp import GradScaler, autocast

def train_catdog_model(model, train_loader, val_loader, num_epochs=10):
    device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
    model = model.to(device)
    
    criterion = nn.CrossEntropyLoss()
    optimizer = torch.optim.AdamW(model.parameters(), lr=0.001, weight_decay=0.01)
    scheduler = torch.optim.lr_scheduler.CosineAnnealingLR(optimizer, T_max=num_epochs)
    
    # 混合精度训练
    scaler = GradScaler()
    
    history = {'train_loss': [], 'train_acc': [], 'val_loss': [], 'val_acc': []}
    best_val_acc = 0.0
    
    for epoch in range(num_epochs):
        # 训练阶段
        model.train()
        train_running_loss = 0.0
        train_correct = 0
        train_total = 0
        
        train_bar = tqdm(train_loader, desc=f'Epoch {epoch+1}/{num_epochs} [Train]')
        for inputs, labels in train_bar:
            inputs, labels = inputs.to(device), labels.to(device)
            
            optimizer.zero_grad()
            
            # 使用混合精度
            with autocast():
                outputs = model(inputs)
                loss = criterion(outputs, labels)
            
            # 反向传播
            scaler.scale(loss).backward()
            scaler.step(optimizer)
            scaler.update()
            
            # 统计
            train_running_loss += loss.item() * inputs.size(0)
            _, predicted = torch.max(outputs.data, 1)
            train_total += labels.size(0)
            train_correct += (predicted == labels).sum().item()
            
            # 更新进度条
            train_acc = train_correct / train_total
            train_bar.set_postfix({
                'Loss': f'{loss.item():.4f}',
                'Acc': f'{train_acc:.4f}'
            })
        
        # 验证阶段（省略详细代码...）
        scheduler.step()
    
    return history

💡 理论分析题

问题1：卷积层参数计算

假设有一个卷积层，输入特征图尺寸为32×32×3，使用64个5×5的卷积核，步长为1，填充为2。请计算：

  1. 该卷积层的参数数量
  2. 输出特征图的尺寸
  3. 如果添加批归一化层，额外增加多少参数？
📝 详细解答

1. 参数数量计算：

  • 每个卷积核参数：5×5×3 = 75
  • 偏置参数：1个/卷积核
  • 总参数：64×(75+1) = 4,864个

2. 输出尺寸：

  • 公式：(W-F+2P)/S + 1
  • 计算：(32-5+2×2)/1 + 1 = 32
  • 输出尺寸：32×32×64

3. 批归一化参数：

  • γ参数：64个（缩放）
  • β参数：64个（偏移）
  • 总计：128个额外参数

🔬 实验设计题

问题2：模型改进实验

基于MNIST数据集，设计实验比较以下几种改进方法的效果：

  1. 增加Dropout层（比较不同dropout率：0.2, 0.5, 0.8）
  2. 使用不同的激活函数（ReLU, LeakyReLU, ELU）
  3. 添加批归一化层
  4. 使用不同的优化器（SGD, Adam, AdamW）

要求：设计实验方案，预测结果，并解释原因。

🔬 实验设计方案
1. Dropout率实验
  • 预测：0.5效果最好，0.8过度正则化，0.2正则化不足
  • 原因：适度的dropout能防止过拟合，过高会影响学习能力
2. 激活函数实验
  • 预测：ReLU > LeakyReLU > ELU
  • 原因：MNIST相对简单，ReLU的简单性和效率优势明显
3. 批归一化实验
  • 预测：显著提升收敛速度和最终精度
  • 原因：稳定训练过程，允许使用更高学习率
4. 优化器实验
  • 预测：Adam > AdamW > SGD

class EarlyStopping:
    def __init__(self, patience=7, min_delta=0):
        self.patience = patience
        self.min_delta = min_delta
        self.counter = 0
        self.best_loss = float('inf')
    
    def __call__(self, val_loss):
        if val_loss < self.best_loss - self.min_delta:
            self.best_loss = val_loss
            self.counter = 0
        else:
            self.counter += 1
        return self.counter >= self.patience
